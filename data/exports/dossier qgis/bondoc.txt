<script>
    // Initialisation de la carte centrée sur l'IDF
    var map = L.map('map').setView([48.8566, 2.3522], 9);

    // Fond de carte OpenStreetMap
    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
        attribution: '© OpenStreetMap contributors',
        maxZoom: 18
    }).addTo(map);

    // Couche WMS depuis GeoServer
    var accidentsLayer = L.tileLayer.wms('http://localhost:8080/geoserver/cite/wms', {
        layers: 'cite:carte_accidents_idf',
        format: 'image/png',
        transparent: true,
        version: '1.1.0',
        styles: 'cite:accidents_idf_style',
        opacity: 0.85
    }).addTo(map);

    // Panneau d'information au survol
    var info = L.control({position: 'topright'});

    info.onAdd = function(map) {
        this._div = L.DomUtil.create('div', 'info');
        this.update();
        return this._div;
    };

    info.update = function(props) {
        this._div.innerHTML = '<h3>Accidents IDF 2024</h3>' +
            (props ? 
                '<b>' + props.nom_departement + '</b><br>' +
                props.nombre_accidents + ' accidents<br>' +
                '<small>Cliquez pour plus de détails</small>'
            : 'Survolez un département');
    };

    info.addTo(map);

    // Légende
    var legend = L.control({position: 'bottomright'});

    legend.onAdd = function(map) {
        var div = L.DomUtil.create('div', 'legend');
        div.innerHTML = '<h4>Nombre d\'accidents</h4>';
        
        var classes = [
            { color: '#800026', label: 'Très élevé > 4 000' },
            { color: '#E31A1C', label: 'Élevé 2 000 - 4 000' },
            { color: '#FC4E2A', label: 'Moyen 1 000 - 2 000' },
            { color: '#FED976', label: 'Faible < 1 000' }
        ];

        classes.forEach(function(c) {
            div.innerHTML +=
                '<i style="background:' + c.color + '"></i>' + c.label + '<br>';
        });

        div.innerHTML += '<br><small>Source : data.gouv.fr</small>';
        return div;
    };

    legend.addTo(map);

    // Clic sur la carte pour info détaillée
    map.on('click', function(e) {
        var url = 'http://localhost:8080/geoserver/cite/wms?' +
            'SERVICE=WMS&VERSION=1.1.1&REQUEST=GetFeatureInfo' +
            '&LAYERS=cite:carte_accidents_idf' +
            '&QUERY_LAYERS=cite:carte_accidents_idf' +
            '&INFO_FORMAT=application/json' +
            '&FEATURE_COUNT=1' +
            '&X=50&Y=50' +
            '&SRS=EPSG:4326' +
            '&WIDTH=101&HEIGHT=101' +
            '&BBOX=' + (e.latlng.lng - 0.1) + ',' + 
                       (e.latlng.lat - 0.1) + ',' + 
                       (e.latlng.lng + 0.1) + ',' + 
                       (e.latlng.lat + 0.1);

        fetch(url)
            .then(r => r.json())
            .then(data => {
                if (data.features && data.features.length > 0) {
                    var props = data.features[0].properties;
                    info.update(props);
                }
            });
    });
</script>
